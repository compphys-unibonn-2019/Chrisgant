\section{Introduction}
\begin{frame}
\frametitle{Classical Hall Effect}

\begin{columns}
	\column{6cm}
	\begin{itemize}
		\item Current $I_x$ flowing through a sample with perpendicular magnetic field $B$ is deflected
		\item Hall Voltage:
		\begin{equation*}
		V_H \propto I_x B
		\end{equation*}
		 %$V_H \propto I_x B$ 
		 
	\end{itemize}
		 
	\column[]{6cm}
	\includegraphics[height=0.3\textheight]{classical-hall}
\end{columns}
\begin{columns}
	\column[]{12cm}
	\begin{itemize}
		 \item current density $\boldsymbol{J} = \sigma \boldsymbol{E}$ \quad
		with $\sigma = \begin{pmatrix} \sigma_{xx} & \sigma_{xy}\\ -\sigma_{xy} & \sigma_{xx}\end{pmatrix}$
		
		\item resistivity $\rho = \sigma^{-1} = \begin{pmatrix} \rho_{xx} & \rho_{xy}\\ -\rho_{xy} & \rho_{yy}\end{pmatrix}$
		
		\item classical result: $\rho_{xx} = const.$,\quad $\rho_{xy} = \dfrac{B}{ne}$
	\end{itemize}
\end{columns}	 
	
		
	
\end{frame}

\section{Integer Quantum Hall Effect}
\begin{frame}
\frametitle{Integer Quantum Hall Effect}
	\begin{columns}
		\column[]{12cm}
		\begin{itemize}
			\item electrons in 2d material at low temperatures and high magnetic field\\\ \\
		\end{itemize}
	\end{columns}

	\begin{columns}
	\column[]{6cm}
	\includegraphics[height=0.58\textheight]{integer-qhe}
	
	\column[]{6cm}
	\begin{itemize}
		%\item electrons in 2d material\\at low temperatures and high magnetic field
	
		\item \alert{plateaus} in resistivity at $\rho_{xy} = \dfrac{2\pi\hbar}{e^2} \dfrac{1}{\nu}$, \quad $\nu \in \mathbb{Z}$
		
		\item center of these plateaus at $B = \dfrac{2\pi\hbar n}{\nu e} = \dfrac{n}{\nu}\Phi_0$\\
		where $n = \frac{\mathcal{N}}{A}$
		
		\item by comparison these are the values when the first $\nu$ Landau Levels are filled completely

	\end{itemize}
	\end{columns}
\end{frame}


\begin{frame}
\frametitle{Where does this come from?}
\begin{itemize}
	\begin{columns}
		\column[]{11cm}
		\item compute current $\boldsymbol{I} = -e\boldsymbol{\dot{x}}$, \quad
		with velocity $m\boldsymbol{\dot{x}} = \boldsymbol{p} + e \boldsymbol{A}$
		\begin{equation*}
		\Rightarrow \boldsymbol{I} = -\frac{e}{m} \sum_{filled\ states} \bra{\Psi}{-i\hbar\nabla + e\boldsymbol{A}}\ket{\Psi}
		\end{equation*}
		%$\Rightarrow$ $\boldsymbol{I} = -\frac{e}{m} \sum_{filled\ states} \bra{\Psi}{-i\hbar\nabla + e\boldsymbol{A}}\ket{\Psi}$ 
		\pause
		\vspace{-0.3cm}
		\item components when $\nu$ Landau Levels are filled using the properties of the harmonic oscillator:\\
		$I_x = -\frac{e}{m} \sum_{n=1}^{\nu}\sum_{k} \bra{\Psi_{n,k}}{-i\hbar \partial_x}\ket{\Psi_{n,k}} = 0$
		$I_y = -\frac{e}{m} \sum_{n=1}^{\nu}\sum_{k} \bra{\Psi_{n,k}}{-i\hbar \partial_y + exB}\ket{\Psi_{n,k}} 
		%= -e \nu \sum_{k} \frac{E}{B} 
		= -\frac{e\nu EA}{\Phi_0}$
		%$\phantom{I_y} = -e \nu \sum_{k} \frac{E}{B} = -\frac{e\nu EA}{\Phi_0}$
		\pause
		\item That means we have:
		$\boldsymbol{E} = \begin{pmatrix} E\\ 0 \end{pmatrix} \quad\Rightarrow\quad \boldsymbol{J} = \begin{pmatrix}
		0\\
		-e\nu E /\phi_0
		\end{pmatrix}$\\\ \\
		i.e. $\sigma_{xx} = 0$ \ and \ $\sigma_{xy} = \dfrac{e\nu}{\Phi_0} \quad\Rightarrow\quad $\Ovalbox{$\rho_{xx} = 0$ \ and \ $\rho_{xy} = -\dfrac{2\pi\hbar}{e^2\nu}$}
	\end{columns}
	
	
	
\end{itemize}

\end{frame}
